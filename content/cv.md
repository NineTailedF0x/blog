function switchCvLang(lang) {
  document.querySelectorAll('[data-lang]').forEach(function(el) {
    var show = el.getAttribute('data-lang') === lang;
    el.style.display = show ? '' : 'none';
    if (show) {
      el.style.animation = 'none';
      el.offsetHeight;
      el.style.animation = '';
    }
  });
  document.querySelectorAll('.cv-lang-btn').forEach(function(btn) {
    btn.classList.toggle('active', btn.getAttribute('data-target') === lang);
  });
}
</script>
